resource factorial()
 procedure fact(k: int) returns f: int
      
  if k < 0 -> f := -1
      
   [] k = 0 or k = 1 -> f := 1
      
   [] k > 1 -> f := k * fact(k - 1)
      
      
  fi
   
end fact
   var n: int
   writes("Cuántos factoriales hay? "); read(n)
   write("a")
   fa i := 1 to n ->
      write(i, "el factorial es ", fact(i))
   af
end factorial
